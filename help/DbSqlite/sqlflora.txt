#SQL ID ID = DTVOL
create table dtvol(
  plantname      text     not null,
  recordnumber   integer  not null,
  volnr          integer  not null,
  pname          text     not null,
  isspecflag     boolean  not null,
  genus          integer  not null
  primary key (plantname, recordnumber, volnr)
  );
#END

#SQL ID = IDX1DTVOL
create index i1dtvol on dtvol(volnr, plantname);
#END

#SQL ID ID = DTPLANTNAME
create table dtplantname(
  plantname      text     not null,
  pname          text     not null,
  primary key (plantname, pname)
  );
#END

#SQL ID = IDX1DTPLANTNAME
create index i1dtplantname on dtplantname(pname, plantname);
#END
